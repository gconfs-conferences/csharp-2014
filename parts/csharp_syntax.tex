\begin{frame}[fragile]
  \frametitle{Manipulation}

  \begin{center}{\large Nombres}\end{center}
  \begin{csharpcode*}{fontsize=\scriptsize}
    int[] results = new int[2];
    int a = 4;
    bool vrai = true;
    bool faux = false;
    int b = 6;
    a = -1;
    results[0] = a + b; // [0] = 5
    b = 3;
    results[1] = a * b; // [1] = -3
  \end{csharpcode*}

  \pause

  \begin{center}{\large Strings}\end{center}
  \begin{csharpcode*}{fontsize=\scriptsize}

  \end{csharpcode*}
\end{frame}
